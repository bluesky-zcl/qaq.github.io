dijkstra

该算法用的是贪心。

分两个集合,刚开始只有起点在A集合，其余在B集合.

然后A集合找集合中成员与B集合成员的边中的最小的，把那个成员也发展成A集合的成员

用d[i]来记录从起点到i的最短距离，

用f[i]了表示该节点是否在A集合（true表示在A集合）

每次把一个成员扩展到A集合中都有更新B集合成员中的d[i]，具体看代码

时间复杂度：O(n2)时间复杂度：O(n^2)时间复杂度：O(n2)

#include <iostream>
using namespace std;
int a[1010][1010];
int d[1010];
bool f[1010];
int n,m;
void dij(int x){
    int i,j;
    int minn;
    for(i=1;i<n;i++){//只有n-1个数（除去起点）
        minn=1e9;int c=0;
        for(j=2;j<=n;j++){//找出最小的
            if(f[j]==false&&minn>d[j]){
                c=j;minn=d[j];
            }
        }
        f[c]=true;
        for(j=2;j<=n;j++)
        if(f[j]==false) d[j]=min(d[j],d[c]+a[c][j]);//更新B集合中的d[i]
    }
}
int main(){
    cin>>n>>m;
    int x,y,z;

        a[x][y]=z;a[y][x]=z;
    }
    for(i=1;i<=n;i++) d[i]=a[1][i];
    f[1]=true;//把1加入集合A
    dij(1);//起点是一

dijkstra堆优化

dijkstra堆优化是以小根堆/优先队列来实现的
思想就是在原有的dijkstra的基础下，在寻找周围最小点的时候，
用堆就可以在O(log n)的时间找出最小的可到达点。

堆优化的主要思想就是使用一个优先队列（就是每次弹出的元素一定是整个队列中最小的元素）来代替最近距离的查找，用邻接表代替邻接矩阵，这样可以大幅度节约时间开销。

在这里有几个细节需要处理：

首先来讲，优先队列的数据类型应该是怎样的呢？
我们知道优先队列应该用于快速寻找距离最近的点。由于优先队列只是将最小的那个元素排在前面，因此我们应该定义一种数据类型，使得它包含该节点的编号以及该节点当前与起点的距离。

我们应该在什么时候对队列进行操作呢？
队列操作的地方，首先就是搜索刚开始，要为起点赋初始值，此时必须将起点加入优先队列中。该队列元素的节点编号为起点的编号，该节点当前与起点的距离为0。

那么如果一个节点到起点的最短距离通过其他的运算流程发生了变化，那么如何处理队列中的那个已经存入的元素？
事实上，你不需要理会队列中的元素，而是再存入一个就行了。因为如果要发生变化，只能将节点与起点之间的距离变得更小，而优先队列恰好是先让最小的那个弹出。

因此，轮到某一个队列元素弹出的时候，如果有多个元素的节点编号相同，那么被弹出的一定是节点编号最小的一个。等到后面再遇到这个节点编号的时候，我们只需要将它忽略掉就行了。

时间复杂度O(n logn)

#include <bits/stdc++.h>
#define MAXN 1000010
using namespace std;
struct ff{
    int v,c;
    ff(int _v=0,int _c=0):v(_v),c(_c){}
    bool operator <(const ff &r)const{
        return c>r.c;
    }
};
struct kkk{
    int v,cost;
    kkk(int _v=0,int _cost=0):v(_v),cost(_cost){}
};
vector<kkk> E[MAXN];
bool vis[MAXN];
int d[MAXN];
void dij(int n,int start){//点的编号从1开始
    memset(vis,false,sizeof(vis));
    int i;
    for(i=1;i<=n;i++) d[i]=1e9;
    priority_queue<ff> que;//优先队列
    while(!que.empty()) que.pop();
    d[start]=0;
    que.push(ff(start,0));
    ff tmp;
    while(!que.empty()){
        tmp=que.top();
        que.pop();
        int u=tmp.v;
        if(vis[u]) continue;
        vis[u]=true;
        for(i=0;i<E[u].size();i++){
            int v=E[tmp.v][i].v;
            int cost=E[u][i].cost;
            if(!vis[v]&&d[v]>d[u]+cost){
                d[v]=d[u]+cost;
                que.push(ff(v,d[v]));
            }
        }
    }
}
void add(int u,int v,int w){
    E[u].push_back(kkk(v,w));
}
int main(){
    int i;
    for(i=0;i<=MAXN;i++) if(!E[i].empty()) E[i].clear();
    int n,e;
    scanf("%d%d",&n,&e);
    int u,v,w;
    for(i=1;i<=e;i++) scanf("%d%d%d",&u,&v,&w),add(u,v,w);
    dij(n,1);
    for(i=1;i<=n;i++)printf("%d ",d[i]);
    printf("\n");
    return 0;
}

queue<int> q;
void add(int x,int y,int z){
    e[++c].x=x;
    e[c].y=y;
    e[c].z=z;
    next[c]=frist[x];
    frist[x]=c;
    return;
}
void spfa(int s){
    memset (d,0x3f,sizeof (d));
    d[s]=0;
    f[s]=true;
    q.push(s);
    int x,i,t;
    while(!q.empty()){
        x=q.front();
        q.pop();
        f[x]=false;
        for (i=frist[x];i;i=next[i]){
            t=e[i].y;
            if(d[t]>d[x]+e[i].z){
                d[t]=d[x]+e[i].z;
                if (f[t]==false){
                    f[t]=true;
                    q.push(t);
                }
            }
        }
    }
    return;
}
